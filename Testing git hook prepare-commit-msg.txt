That file should automatically add a signed-by-line to each commmit
as described in the following stackoverflow posting 
https://stackoverflow.com/questions/15015894/git-add-signed-off-by-line-using-format-signoff-not-working/46536244#46536244

=========================
     #!/bin/sh
     
     NAME=$(git config user.name)
     EMAIL=$(git config user.email)
     
     if [ -z "$NAME" ]; then
         echo "empty git config user.name"
         exit 1
     fi
     
     if [ -z "$EMAIL" ]; then
         echo "empty git config user.email"
         exit 1
     fi
     
     git interpret-trailers --if-exists doNothing --trailer \
         "Signed-off-by: $NAME <$EMAIL>" \
         --in-place "$1"
